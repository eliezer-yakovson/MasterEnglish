import { useDeferredValue, useEffect, useMemo, useState } from "react";

import { createApiClient, defaultConfig } from "./api";

const CONFIG_STORAGE_KEY = "master-english-ui-config";
const SESSION_STORAGE_KEY = "master-english-ui-session";

const emptyProgress = {
  total_words: 0,
  words_mastered: 0,
  words_in_progress: 0,
  streak_days: 0,
  last_session_date: null,
  total_score: 0,
};

const initialRegisterForm = {
  name: "",
  email: "",
  password: "",
  role: "user",
};

const initialLoginForm = {
  email: "",
  password: "",
};

const initialManualForm = {
  word: "",
  context: "",
};

const initialCaptureForm = {
  word: "",
  context: "",
  lang: "he",
};

function loadJsonStorage(key, fallbackValue) {
  const raw = localStorage.getItem(key);
  if (!raw) {
    return fallbackValue;
  }

  try {
    return JSON.parse(raw);
  } catch {
    return fallbackValue;
  }
}

function shortDate(value) {
  if (!value) {
    return "עדיין אין";
  }

  try {
    return new Intl.DateTimeFormat("he-IL", { dateStyle: "medium" }).format(new Date(value));
  } catch {
    return value;
  }
}

function splitImportWords(text) {
  return Array.from(
    new Set(
      text
        .split(/[\n,;]+/)
        .map((item) => item.trim().toLowerCase())
        .filter(Boolean),
    ),
  );
}

function App() {
  const [config, setConfig] = useState(() => loadJsonStorage(CONFIG_STORAGE_KEY, defaultConfig));
  const [session, setSession] = useState(() => loadJsonStorage(SESSION_STORAGE_KEY, null));
  const [showSettings, setShowSettings] = useState(false);
  const [authMode, setAuthMode] = useState("login");
  const [loginForm, setLoginForm] = useState(initialLoginForm);
  const [registerForm, setRegisterForm] = useState(initialRegisterForm);
  const [activeView, setActiveView] = useState(session?.role === "admin" ? "admin" : "learner");
  const [statusMessage, setStatusMessage] = useState("");
  const [errorMessage, setErrorMessage] = useState("");
  const [busyKey, setBusyKey] = useState("");

  const [progress, setProgress] = useState(emptyProgress);
  const [library, setLibrary] = useState({ total: 0, items: [] });
  const [librarySearch, setLibrarySearch] = useState("");
  const deferredLibrarySearch = useDeferredValue(librarySearch);
  const [manualForm, setManualForm] = useState(initialManualForm);
  const [bulkImportText, setBulkImportText] = useState("");
  const [importSummary, setImportSummary] = useState(null);

  const [trainingCount, setTrainingCount] = useState(6);
  const [trainingSession, setTrainingSession] = useState(null);
  const [trainingIndex, setTrainingIndex] = useState(0);
  const [trainingStartedAt, setTrainingStartedAt] = useState(null);
  const [trainingSummary, setTrainingSummary] = useState([]);

  const [captureForm, setCaptureForm] = useState(initialCaptureForm);
  const [capturePreview, setCapturePreview] = useState(null);

  const [adminOverview, setAdminOverview] = useState(null);
  const [adminSearch, setAdminSearch] = useState("");
  const [systemWords, setSystemWords] = useState({ total: 0, items: [] });
  const [systemWordsSearch, setSystemWordsSearch] = useState("");
  const deferredSystemWordsSearch = useDeferredValue(systemWordsSearch);

  const api = useMemo(
    () => createApiClient(config, session?.access_token),
    [config, session?.access_token],
  );

  const currentTrainingItem = trainingSession?.queue?.[trainingIndex] ?? null;

  const filteredAdminUsers = useMemo(() => {
    const users = adminOverview?.users || [];
    const term = adminSearch.trim().toLowerCase();

    if (!term) {
      return users;
    }

    return users.filter((user) =>
      [user.name, user.email, user.user_id, user.role, user.status]
        .filter(Boolean)
        .some((value) => String(value).toLowerCase().includes(term)),
    );
  }, [adminOverview, adminSearch]);

  useEffect(() => {
    localStorage.setItem(CONFIG_STORAGE_KEY, JSON.stringify(config));
  }, [config]);

  useEffect(() => {
    if (session) {
      localStorage.setItem(SESSION_STORAGE_KEY, JSON.stringify(session));
      return;
    }

    localStorage.removeItem(SESSION_STORAGE_KEY);
  }, [session]);

  useEffect(() => {
    if (!session) {
      setActiveView("learner");
      return;
    }

    if (session.role === "admin") {
      setActiveView("admin");
    }
  }, [session]);

  useEffect(() => {
    if (!session || session.role === "admin") {
      return undefined;
    }

    const timeoutId = window.setTimeout(() => {
      refreshLibrary(deferredLibrarySearch);
    }, 250);

    return () => window.clearTimeout(timeoutId);
  }, [deferredLibrarySearch, session?.user_id]);

  useEffect(() => {
    if (session?.role !== "admin") {
      return undefined;
    }

    const timeoutId = window.setTimeout(() => {
      refreshSystemWords(deferredSystemWordsSearch);
    }, 250);

    return () => window.clearTimeout(timeoutId);
  }, [deferredSystemWordsSearch, session?.user_id]);

  async function withBusy(key, action) {
    setBusyKey(key);
    setErrorMessage("");
    setStatusMessage("");

    try {
      await action();
    } catch (error) {
      setErrorMessage(error.message || "אירעה שגיאה לא צפויה.");
    } finally {
      setBusyKey("");
    }
  }

  async function refreshProgress(targetSession = session) {
    if (!targetSession?.user_id || targetSession.role === "admin") {
      return;
    }

    const payload = await api.getProgress(targetSession.user_id);
    setProgress(payload);
  }

  async function refreshLibrary(searchTerm = librarySearch, targetSession = session) {
    if (!targetSession?.user_id || targetSession.role === "admin") {
      return;
    }

    const payload = await api.getLibrary(targetSession.user_id, searchTerm, 100);
    setLibrary(payload);
  }

  async function refreshAdminOverview() {
    if (session?.role !== "admin") {
      return;
    }

    const payload = await api.getAdminOverview();
    setAdminOverview(payload);
  }

  async function refreshSystemWords(searchTerm = systemWordsSearch) {
    if (session?.role !== "admin") {
      return;
    }

    const payload = await api.getSystemWords(searchTerm, 200);
    setSystemWords(payload);
  }

  async function hydrateAfterLogin(nextSession) {
    if (nextSession.role === "admin") {
      const nextApi = createApiClient(config, nextSession.access_token);
      const [overview, wordsPayload] = await Promise.all([
        nextApi.getAdminOverview(),
        nextApi.getSystemWords("", 200)
      ]);
      setAdminOverview(overview);
      setSystemWords(wordsPayload);
      setProgress(emptyProgress);
      setLibrary({ total: 0, items: [] });
      return;
    }

    const nextApi = createApiClient(config, nextSession.access_token);
    const [nextProgress, nextLibrary] = await Promise.all([
      nextApi.getProgress(nextSession.user_id),
      nextApi.getLibrary(nextSession.user_id, "", 100),
    ]);
    setProgress(nextProgress);
    setLibrary(nextLibrary);
    setAdminOverview(null);
  }

  async function handleLogin(event) {
    event.preventDefault();

    await withBusy("login", async () => {
      const result = await api.login(loginForm);
      const nextSession = {
        ...result,
        email: loginForm.email,
      };

      setSession(nextSession);
      await hydrateAfterLogin(nextSession);
      setLoginForm(initialLoginForm);
      setStatusMessage(result.role === "admin" ? "כניסת מנהל הצליחה." : "נכנסת בהצלחה לאזור האישי.");
    });
  }

  async function handleRegister(event) {
    event.preventDefault();

    await withBusy("register", async () => {
      const result = await api.register(registerForm);
      setRegisterForm(initialRegisterForm);
      setAuthMode("login");
      setStatusMessage(`נרשם משתמש חדש בהצלחה. מזהה: ${result.user_id}`);
    });
  }

  async function handleLogout() {
    await withBusy("logout", async () => {
      if (session?.access_token) {
        await api.logout();
      }

      setSession(null);
      setProgress(emptyProgress);
      setLibrary({ total: 0, items: [] });
      setAdminOverview(null);
      setTrainingSession(null);
      setTrainingSummary([]);
      setCapturePreview(null);
      setStatusMessage("היציאה בוצעה בהצלחה.");
    });
  }

  async function handleManualSave(event) {
    event.preventDefault();

    await withBusy("manual-save", async () => {
      const result = await api.ingestWord({
        word: manualForm.word,
        user_id: session.user_id,
        source: "manual",
        context: manualForm.context || undefined,
      });

      setManualForm(initialManualForm);
      await Promise.all([refreshLibrary(""), refreshProgress()]);
      setStatusMessage(
        result.status === "duplicate"
          ? "המילה כבר קיימת בספרייה, אבל נשמר לך המידע המועשר להצגה."
          : `המילה נשמרה בהצלחה. תרגום: ${result.enriched_word.translation}`,
      );
    });
  }

  async function handleBulkImport(event) {
    event.preventDefault();

    await withBusy("bulk-import", async () => {
      const words = splitImportWords(bulkImportText);
      if (!words.length) {
        throw new Error("צריך להדביק לפחות מילה אחת לייבוא.");
      }

      const summary = {
        total: words.length,
        created: 0,
        duplicates: 0,
        failed: [],
      };

      for (const word of words) {
        try {
          const duplicate = await api.checkDuplicate(word, session.user_id);
          if (duplicate.exists) {
            summary.duplicates += 1;
            continue;
          }

          await api.ingestWord({
            word,
            user_id: session.user_id,
            source: "import",
          });
          summary.created += 1;
        } catch (error) {
          summary.failed.push({ word, reason: error.message || "שגיאה" });
        }
      }

      setImportSummary(summary);
      setBulkImportText("");
      await Promise.all([refreshLibrary(""), refreshProgress()]);
      setStatusMessage("הייבוא הסתיים.");
    });
  }

  async function handleTrainingStart() {
    await withBusy("training-start", async () => {
      const result = await api.getTrainingQueue(session.user_id, trainingCount);
      setTrainingSession(result);
      setTrainingIndex(0);
      setTrainingSummary([]);
      setTrainingStartedAt(Date.now());
      if (!result.queue.length) {
        setStatusMessage("אין כרגע מילים שממתינות לתרגול.");
      } else {
        setStatusMessage(`נטען סבב תרגול עם ${result.queue.length} מילים.`);
      }
    });
  }

  async function handleTrainingAnswer(answer) {
    if (!currentTrainingItem) {
      return;
    }

    await withBusy(`training-${answer}`, async () => {
      const startedAt = trainingStartedAt || Date.now();
      const response = await api.submitTrainingResult({
        session_id: trainingSession.session_id,
        word_id: currentTrainingItem.word_id,
        user_id: session.user_id,
        result: answer,
        time_taken_ms: Date.now() - startedAt,
      });

      setTrainingSummary((current) => [
        ...current,
        {
          word: currentTrainingItem.word,
          answer,
          next_review_date: response.next_review_date,
          new_stage: response.new_stage,
        },
      ]);

      const nextIndex = trainingIndex + 1;
      if (nextIndex >= trainingSession.queue.length) {
        setTrainingSession(null);
        setTrainingIndex(0);
        setTrainingStartedAt(null);
        await Promise.all([refreshProgress(), refreshLibrary("")]);
        setStatusMessage("סבב התרגול הושלם והנתונים עודכנו.");
        return;
      }

      setTrainingIndex(nextIndex);
      setTrainingStartedAt(Date.now());
    });
  }

  async function handleCapturePreview(event) {
    event.preventDefault();

    await withBusy("capture-preview", async () => {
      const [enriched, duplicate] = await Promise.all([
        api.enrichWord(captureForm.word, captureForm.lang),
        api.checkDuplicate(captureForm.word, session.user_id),
      ]);
      setCapturePreview({ enriched, duplicate });
      setStatusMessage(duplicate.exists ? "המילה כבר קיימת אצל המשתמש." : "המידע המועשר נטען בהצלחה.");
    });
  }

  async function handleCaptureSave() {
    await withBusy("capture-save", async () => {
      const result = await api.ingestWord({
        word: captureForm.word,
        user_id: session.user_id,
        source: "extension",
        context: captureForm.context || undefined,
      });

      await Promise.all([refreshLibrary(""), refreshProgress()]);
      setCapturePreview((current) =>
        current
          ? { ...current, duplicate: { exists: true, word_id: result.word_id } }
          : current,
      );
      setStatusMessage(
        result.status === "duplicate"
          ? "המילה כבר הייתה קיימת בספרייה."
          : "אירוע הלכידה נשמר ונכנס למסלול האישי.",
      );
    });
  }

  async function handleAdminAction(targetUser, actionType, newRole) {
    await withBusy(`admin-${actionType}-${targetUser.user_id}`, async () => {
      const payload = {
        admin_id: session.user_id,
        action_type: actionType,
        target_user_id: targetUser.user_id,
      };

      if (newRole) {
        payload.new_role = newRole;
      }

      const result = await api.adminAction(targetUser.user_id, payload);
      await refreshAdminOverview();
      setStatusMessage(result.message || "הפעולה בוצעה בהצלחה.");
    });
  }

  async function handleAdminDelete(targetUser) {
    await withBusy(`admin-delete-${targetUser.user_id}`, async () => {
      const result = await api.deleteUser(targetUser.user_id);
      await refreshAdminOverview();
      setStatusMessage(result.message || "המשתמש נמחק.");
    });
  }

  async function handleSystemDedup() {
    await withBusy("admin-dedup", async () => {
      const result = await api.dedupSystemWords();
      await refreshSystemWords();
      setStatusMessage(`בדיקת כפילויות הסתיימה: נמחקו ${result.removed} מילים מתוך המאגר המשותף.`);
    });
  }

  const isLoggedIn = Boolean(session?.access_token);
  const isAdmin = session?.role === "admin";

  if (!isLoggedIn) {
    return (
      <div className="auth-layout">
        <div className="auth-card">
          <div className="brand">
            <span className="brand-dot"></span>
            <h1>Master English</h1>
          </div>
          <h2>ברוכים הבאים</h2>
          <p className="auth-subtitle">מערכת חכמה לניהול ולמידת אוצר מילים</p>
          
          <div className="segmented-control">
            <button className={authMode === "login" ? "active" : ""} onClick={() => setAuthMode("login")}>
              התחברות
            </button>
            <button className={authMode === "register" ? "active" : ""} onClick={() => setAuthMode("register")}>
              הרשמה
            </button>
          </div>

          <div className="message-stack">
            {statusMessage ? <div className="message success">{statusMessage}</div> : null}
            {errorMessage ? <div className="message error">{errorMessage}</div> : null}
          </div>

          {authMode === "login" ? (
            <form className="auth-form" onSubmit={handleLogin}>
              <div className="input-group">
                <label>אימייל</label>
                <input type="email" value={loginForm.email} onChange={(e) => setLoginForm(curr => ({ ...curr, email: e.target.value }))} placeholder="yael@example.com" required />
              </div>
              <div className="input-group">
                <label>סיסמה</label>
                <input type="password" value={loginForm.password} onChange={(e) => setLoginForm(curr => ({ ...curr, password: e.target.value }))} placeholder="••••••••" required />
              </div>
              <button className="btn-primary" type="submit" disabled={busyKey === "login"}>
                {busyKey === "login" ? "מתחבר..." : "כניסה למערכת"}
              </button>
            </form>
          ) : (
            <form className="auth-form" onSubmit={handleRegister}>
              <div className="input-group">
                <label>שם מלא</label>
                <input value={registerForm.name} onChange={(e) => setRegisterForm(curr => ({ ...curr, name: e.target.value }))} placeholder="יעל כהן" required />
              </div>
              <div className="input-group">
                <label>אימייל</label>
                <input type="email" value={registerForm.email} onChange={(e) => setRegisterForm(curr => ({ ...curr, email: e.target.value }))} placeholder="yael@example.com" required />
              </div>
              <div className="input-group">
                <label>סיסמה</label>
                <input type="password" value={registerForm.password} onChange={(e) => setRegisterForm(curr => ({ ...curr, password: e.target.value }))} placeholder="לפחות 6 תווים" required />
              </div>
              <div className="input-group">
                <label>תפקיד</label>
                <select value={registerForm.role} onChange={(e) => setRegisterForm(curr => ({ ...curr, role: e.target.value }))}>
                  <option value="user">משתמש לומד</option>
                  <option value="admin">מנהל (Admin)</option>
                </select>
              </div>
              <button className="btn-primary" type="submit" disabled={busyKey === "register"}>
                {busyKey === "register" ? "יוצר מחשבון..." : "צור חשבון חדש"}
              </button>
            </form>
          )}

          <div className="auth-footer">
            <button className="btn-link" type="button" onClick={() => setShowSettings(!showSettings)}>
              {showSettings ? 'הסתר הגדרות מתקדמות' : 'הגדרות מתקדמות'}
            </button>
            
            {showSettings && (
              <div className="config-box">
                <label>
                  <span>Admin URL</span>
                  <input value={config.adminBaseUrl} onChange={(e) => setConfig(curr => ({ ...curr, adminBaseUrl: e.target.value }))} />
                </label>
                <label>
                  <span>DATA 1 URL</span>
                  <input value={config.data1BaseUrl} onChange={(e) => setConfig(curr => ({ ...curr, data1BaseUrl: e.target.value }))} />
                </label>
                <label>
                  <span>DATA 2 URL</span>
                  <input value={config.data2BaseUrl} onChange={(e) => setConfig(curr => ({ ...curr, data2BaseUrl: e.target.value }))} />
                </label>
              </div>
            )}
          </div>
        </div>
      </div>
    );
  }

  return (
    <div className="app-layout">
      <nav className="main-navbar">
        <div className="nav-container">
          <div className="nav-brand">
            <span className="brand-dot"></span>
            <span className="brand-text">Master English</span>
          </div>
          <div className="nav-links desktop-links">
            {!isAdmin && (
              <>
                <button className={activeView === "learner" ? "active" : ""} onClick={() => setActiveView("learner")}>אזור למידה</button>
                <button className={activeView === "capture" ? "active" : ""} onClick={() => setActiveView("capture")}>לכידת מילים</button>
              </>
            )}
            {isAdmin && (
              <button className={activeView === "admin" ? "active" : ""} onClick={() => setActiveView("admin")}>פאנל ניהול</button>
            )}
          </div>
          <div className="nav-user">
            <span className="user-badge">{session?.email?.charAt(0).toUpperCase() || 'U'}</span>
            <button className="btn-logout" onClick={handleLogout} disabled={busyKey === "logout"}>
              יציאה
            </button>
          </div>
        </div>
      </nav>

      <main className="main-content">
        <div className="message-stack container">
          {statusMessage ? <div className="message success">{statusMessage}</div> : null}
          {errorMessage ? <div className="message error">{errorMessage}</div> : null}
        </div>

        <div className="container">
          {activeView === "learner" && (
            <LearnerView
              progress={progress}
              library={library}
              librarySearch={librarySearch}
              setLibrarySearch={setLibrarySearch}
              manualForm={manualForm}
              setManualForm={setManualForm}
              bulkImportText={bulkImportText}
              setBulkImportText={setBulkImportText}
              importSummary={importSummary}
              trainingCount={trainingCount}
              setTrainingCount={setTrainingCount}
              trainingSession={trainingSession}
              currentTrainingItem={currentTrainingItem}
              trainingIndex={trainingIndex}
              trainingSummary={trainingSummary}
              onManualSave={handleManualSave}
              onTrainingStart={handleTrainingStart}
              onTrainingAnswer={handleTrainingAnswer}
              onBulkImport={handleBulkImport}
              refreshProgress={() => withBusy("refresh-progress", () => refreshProgress())}
              refreshLibrary={() => withBusy("refresh-library", () => refreshLibrary(librarySearch))}
              busyKey={busyKey}
            />
          )}

          {activeView === "capture" && (
            <CaptureView
              captureForm={captureForm}
              setCaptureForm={setCaptureForm}
              capturePreview={capturePreview}
              onPreview={handleCapturePreview}
              onSave={handleCaptureSave}
              busyKey={busyKey}
            />
          )}

          {activeView === "admin" && (
            <AdminView
              overview={adminOverview}
              search={adminSearch}
              setSearch={setAdminSearch}
              users={filteredAdminUsers}
              systemWords={systemWords}
              systemWordsSearch={systemWordsSearch}
              setSystemWordsSearch={setSystemWordsSearch}
              onDedup={handleSystemDedup}
              onRefresh={() => {
                withBusy("admin-refresh", refreshAdminOverview);
                withBusy("admin-refresh-words", refreshSystemWords);
              }}
              onAction={handleAdminAction}
              onDelete={handleAdminDelete}
              busyKey={busyKey}
            />
          )}
        </div>
      </main>

      <nav className="mobile-bottom-nav">
        {!isAdmin && (
          <>
            <button className={activeView === "learner" ? "active" : ""} onClick={() => setActiveView("learner")}>
               ספרייה 📚
            </button>
            <button className={activeView === "capture" ? "active" : ""} onClick={() => setActiveView("capture")}>
               לכידה ➕
            </button>
          </>
        )}
        {isAdmin && (
          <button className={activeView === "admin" ? "active" : ""} onClick={() => setActiveView("admin")}>
            ניהול ⚙️
          </button>
        )}
      </nav>
    </div>
  );
}

function LearnerView({
  progress,
  library,
  librarySearch,
  setLibrarySearch,
  manualForm,
  setManualForm,
  bulkImportText,
  setBulkImportText,
  importSummary,
  trainingCount,
  setTrainingCount,
  trainingSession,
  currentTrainingItem,
  trainingIndex,
  trainingSummary,
  onManualSave,
  onTrainingStart,
  onTrainingAnswer,
  onBulkImport,
  refreshProgress,
  refreshLibrary,
  busyKey,
}) {
  return (
    <div className="view-grid">
      <div className="view-header">
        <h2>ההתקדמות שלי</h2>
        <button className="btn-secondary" onClick={refreshProgress} disabled={busyKey === "refresh-progress"}>רענון נתונים</button>
      </div>

      <div className="stats-row">
        <MetricCard label="מילים במאגר" value={progress.total_words} color="red" />
        <MetricCard label="נלמדו בהצלחה" value={progress.words_mastered} color="green" />
        <MetricCard label="בתהליך למידה" value={progress.words_in_progress} color="orange" />
        <MetricCard label="רצף ימים" value={progress.streak_days} color="red" />
      </div>

      <div className="cards-grid">
        <div className="content-card training-section">
          <h3>אזור אימון</h3>
          <p className="subtitle">חזרה על מילים מחזקת את הזיכרון לטווח ארוך.</p>
          
          <div className="training-controls">
            <label>
              <span>מספר מילים לאימון:</span>
              <select value={trainingCount} onChange={(e) => setTrainingCount(Number(e.target.value))}>
                <option value={4}>4 מילים</option>
                <option value={6}>6 מילים</option>
                <option value={10}>10 מילים</option>
              </select>
            </label>
            <button className="btn-primary" onClick={onTrainingStart} disabled={busyKey === "training-start"}>
              {busyKey === "training-start" ? "מכין..." : "התחל אימון עכשיו"}
            </button>
          </div>

          {currentTrainingItem && (
            <div className="active-training">
              <div className="progress-badge">מילה {trainingIndex + 1} מתוך {trainingSession.queue.length}</div>
              <h2 className="training-word">{currentTrainingItem.word}</h2>
              <p className="training-desc">{currentTrainingItem.translation}</p>
              
              <div className="training-actions">
                <button className="action-btn success" onClick={() => onTrainingAnswer("correct")} disabled={busyKey.startsWith("training-")}>
                  ידעתי 🎉
                </button>
                <button className="action-btn danger" onClick={() => onTrainingAnswer("incorrect")} disabled={busyKey.startsWith("training-")}>
                  לא זכרתי 😕
                </button>
              </div>
            </div>
          )}

          {trainingSummary.length > 0 && (
            <div className="training-summary-box">
              <h4>סיכום אימון אחרון</h4>
              <div className="summary-list">
                {trainingSummary.map((item, idx) => (
                  <div className="summary-item" key={idx}>
                    <strong>{item.word}</strong>
                    <span className={`status-icon ${item.answer}`}>{item.answer === "correct" ? "✓" : "✗"}</span>
                  </div>
                ))}
              </div>
            </div>
          )}
        </div>

        <div className="content-card add-words-section">
          <h3>הוספת מילים</h3>
          
          <form className="simple-form" onSubmit={onManualSave}>
            <h4>מילה בודדת</h4>
            <div className="input-group">
              <input type="text" value={manualForm.word} onChange={(e) => setManualForm(c => ({...c, word: e.target.value}))} placeholder="למשל: resilient" required />
            </div>
            <div className="input-group">
              <textarea value={manualForm.context} onChange={(e) => setManualForm(c => ({...c, context: e.target.value}))} placeholder="משפט לדוגמה..." rows="2" />
            </div>
            <button className="btn-secondary" type="submit" disabled={busyKey === "manual-save"}>
              {busyKey === "manual-save" ? "מוסיף..." : "הוסף מילה"}
            </button>
          </form>

          <form className="simple-form mt-4" onSubmit={onBulkImport}>
            <h4>דחיפת רשימת מילים</h4>
            <div className="input-group">
              <textarea value={bulkImportText} onChange={(e) => setBulkImportText(e.target.value)} placeholder="הדבק מילים (אחת בכל שורה)..." rows="3" required />
            </div>
            <button className="btn-secondary" type="submit" disabled={busyKey === "bulk-import"}>
              {busyKey === "bulk-import" ? "מייבא..." : "ייבא רשימה"}
            </button>
            {importSummary && (
              <div className="import-result">
                יובאו {importSummary.created} בהצלחה! ({importSummary.duplicates} כפילויות, {importSummary.failed.length} שגיאות).
              </div>
            )}
          </form>
        </div>

        <div className="content-card library-section full-width">
          <div className="library-header">
            <h3>ספריית המילים שלי</h3>
            <span className="library-count">{library.total} פריטים</span>
          </div>
          
          <div className="input-group">
            <input type="text" value={librarySearch} onChange={(e) => setLibrarySearch(e.target.value)} placeholder="חפש מילה או תרגום..." />
          </div>
          
          <div className="library-grid">
            {library.items.length ? (
              library.items.map((item) => (
                <div className="library-card" key={item.word_id}>
                  <div className="card-top">
                    <strong className="word-title">{item.word}</strong>
                    <span className="difficulty-badge">{item.difficulty}</span>
                  </div>
                  <p className="word-translation">{item.translation}</p>
                  <div className="word-stats">
                    <span>שלב {item.knowledge_stage}</span>
                    <span>ציון {item.score}</span>
                  </div>
                </div>
              ))
            ) : (
              <div className="empty-state">אין מילים שמתאימות לחיפוש.</div>
            )}
          </div>
        </div>

      </div>
    </div>
  );
}

function CaptureView({ captureForm, setCaptureForm, capturePreview, onPreview, onSave, busyKey }) {
  return (
    <div className="view-grid">
      <div className="view-header">
        <h2>לכידת מילים מהירה</h2>
      </div>
      
      <div className="capture-layout">
        <div className="content-card">
          <h3>איזו מילה פגשתם?</h3>
          <p className="subtitle">המערכת תעשיר את המילה אוטומטית עם תרגום, הגדרה ומשפטים.</p>
          
          <form className="simple-form capture-form" onSubmit={onPreview}>
            <div className="input-group capture-word">
              <input type="text" value={captureForm.word} onChange={(e) => setCaptureForm(c => ({...c, word: e.target.value}))} placeholder="Type an English word..." required autoFocus />
            </div>
            <div className="input-group">
              <textarea value={captureForm.context} onChange={(e) => setCaptureForm(c => ({...c, context: e.target.value}))} placeholder="משפט ההקשר, איפה קראתם את זה? מומלץ!" rows="2" />
            </div>
            <div className="input-group split">
              <label>שפת תרגום:</label>
              <select value={captureForm.lang} onChange={(e) => setCaptureForm(c => ({...c, lang: e.target.value}))}>
                <option value="he">עברית</option>
                <option value="en">אנגלית</option>
              </select>
            </div>
            <button className="btn-primary large-action" type="submit" disabled={busyKey === "capture-preview"}>
              {busyKey === "capture-preview" ? "מעבד ומעשיר..." : "בדוק והעשר"}
            </button>
          </form>
        </div>

        {capturePreview ? (
          <div className="content-card preview-card">
            <div className="preview-top">
              <h1>{capturePreview.enriched.word}</h1>
              <span className="difficulty-badge orange">{capturePreview.enriched.difficulty_level}</span>
            </div>
            
            <p className="preview-translation">{capturePreview.enriched.translation}</p>
            <p className="preview-definition">{capturePreview.enriched.definition}</p>
            
            <div className="preview-examples">
              <h4>דוגמאות שימוש</h4>
              <ul>
                {capturePreview.enriched.examples.map((ex, i) => <li key={i}>{ex}</li>)}
              </ul>
            </div>

            {capturePreview.duplicate.exists ? (
              <div className="notice-box warning">
                המילה כבר קיימת בספרייה שלכם!
              </div>
            ) : (
              <button className="btn-success large-action mt-4" onClick={onSave} disabled={busyKey === "capture-save"}>
                {busyKey === "capture-save" ? "שומר..." : "הוסף לספרייה שלי ➕"}
              </button>
            )}
          </div>
        ) : (
          <div className="empty-state-card">
            הקלידו מילה והמתינו לקסם... ✨
          </div>
        )}
      </div>
    </div>
  );
}
